----------------------------------------------------------------------------------------------------------------------
_______________________to build project
--------------------------------------------------------------------------------------------------------------------------
sudo docker build -t myapp-next .

----------------------------------------------------------------------------------------------------------------------
_______________________to check things running
--------------------------------------------------------------------------------------------------------------------------
docker ps

----------------------------------------------------------------------------------------------------------------------
_______________________to run the project
--------------------------------------------------------------------------------------------------------------------------

sudo docker run -dp 3000:3000 myapp-next
----------------------------------------------------------------------------------------------------------------------
_______________________clear space
--------------------------------------------------------------------------------------------------------------------------

sudo docker container prune
sudo docker image prune
sudo docker image prune -a
sudo docker volume prune
sudo docker network prune
sudo docker system prune
sudo docker system prune -a
sudo docker system df

----------------------------------------------------------------------------------------------------------------------
_______________________stop all running ports
--------------------------------------------------------------------------------------------------------------------

sudo docker stop $(sudo docker ps -q)

----------------------------------------------------------------------------------------------------------------------
_______________________check running ports
--------------------------------------------------------------------------------------------------------------------
sudo docker ps
